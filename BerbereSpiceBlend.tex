\subsection*{Ingredients}

\begin{itemize}
  \item $ \frac{1}{2} $ cup ground dried chillis
  \item $ \frac{1}{4} $ cup paprika
  \item $ 1 $ tablespoon cayenne pepper
  \item $ 1 $ teaspoon onion powder
  \item $ 1 $ teaspoon ground ginger
  \item $ 1 $ teaspoon cumin
  \item $ 1 $ teaspoon ground coriander
  \item $ 1 $ teaspoon ground cardamom
  \item $ 1 $ teaspoon ground fenugreek
  \item $ \frac{1}{2} $ teaspoon garlic powder
  \item $ \frac{1}{2} $ teaspoon ground cinnamon
  \item $ \frac{1}{2} $ teaspoon ground allspice
  \item $ \frac{1}{2} $ teaspoon ground cloves
  \item $ \frac{1}{4} $ teaspoon ground nutmeg
\end{itemize}

\medskip

\subsection*{Method}

\begin{enumerate}
  \item Place ground chiles, paprika, and cayenne pepper into a bowl; stir. 
  \item Add onion powder, ginger, cumin, coriander, cardamom, fenugreek, garlic powder, cinnamon, allspice, cloves, and nutmeg. 
  \item Gently whisk together until thoroughly mixed.
\end{enumerate}
